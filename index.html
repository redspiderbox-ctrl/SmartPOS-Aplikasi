function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var action = e.parameter.action; if (action === "getData") { return ContentService.createTextOutput(JSON.stringify({ success: true, products: readSheetData(getOrCreateSheet(ss, "Produk")), transactions: readSheetData(getOrCreateSheet(ss, "Transaksi")) })).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ success: false })).setMimeType(ContentService.MimeType.JSON); }
function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var postData = JSON.parse(e.postData.contents); var action = postData.action; if (action === "saveTransaction") { var trx = postData.transaction; getOrCreateSheet(ss, "Transaksi").appendRow([trx.id, trx.date, JSON.stringify(trx.items), trx.subtotal, trx.discount, trx.tax, trx.total, trx.paymentMethod, trx.paymentAmount, trx.change]); var prodData = getOrCreateSheet(ss, "Produk").getDataRange().getValues(); trx.items.forEach(function(item) { for (var i = 1; i < prodData.length; i++) { if (prodData[i][1] === item.product.code) { getOrCreateSheet(ss, "Produk").getRange(i + 1, 6).setValue(Math.max(0, prodData[i][5] - item.quantity)); break; } } }); return ContentService.createTextOutput(JSON.stringify({ success: true })).setMimeType(ContentService.MimeType.JSON); } if (action === "updateProducts") { var sheet = getOrCreateSheet(ss, "Produk"); sheet.clear(); sheet.appendRow(["ID", "Code", "Name", "CostPrice", "Price", "Stock", "Category"]); postData.products.forEach(function(p) { sheet.appendRow([p.id, p.code, p.name, p.costPrice || 0, p.price, p.stock, p.category]); }); return ContentService.createTextOutput(JSON.stringify({ success: true })).setMimeType(ContentService.MimeType.JSON); } }
function getOrCreateSheet(ss, name) { var sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); if(name==="Produk") sheet.appendRow(["ID","Code","Name","CostPrice","Price","Stock","Category"]); else sheet.appendRow(["ID","Date","Items","Subtotal","Discount","Tax","Total","PaymentMethod","PaymentAmount","Change"]); } return sheet; }
