Hubungkan Google Apps Script

Masukkan URL Web App dari Google Apps Script untuk menghubungkan SmartPOS Anda secara aman dengan Google Sheet.

Catatan untuk Pengguna: Jika kolom di atas kosong, SmartPOS secara otomatis berjalan pada Local Mode dengan memanfaatkan penyimpanan lokal browser (Local Storage). Semua fitur tetap berfungsi penuh sebagai simulasi kasir.

Sinkronisasi Indeks ke GitHub

Anda dapat menyinkronkan file ini ke repositori GitHub untuk kebutuhan deployment hosting gratis seperti GitHub Pages.

1. Buat Repositori Baru di GitHub dengan nama smart-pos-gas.
2. Simpan file HTML ini sebagai index.html dan lakukan commit ke repositori Anda.
3. Aktifkan GitHub Pages di pengaturan repositori (Settings > Pages > Build and deployment > Branch: main / root).
4. Web Apps siap diakses publik dan tersambung langsung ke Apps Script API Anda!

Source Code Google Apps Script (GAS)

Salin kode Google Apps Script di bawah ini lalu tempel ke Editor Script Google Sheets Anda (Sesuaikan kolom database). 

/*
  GOOGLE APPS SCRIPT DATABASE ENDPOINT - SMARTPOS (UPDATED WITH COSTPRICE)
  Tempel Kode ini pada editor Apps Script Anda (Ekstensi > Apps Script)
*/

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheetProduk = getOrCreateSheet(ss, "Produk");
  var sheetTransaksi = getOrCreateSheet(ss, "Transaksi");
  
  var action = e.parameter.action;
  
  if (action === "getData") {
    var dataProduk = readSheetData(sheetProduk);
    var dataTransaksi = readSheetData(sheetTransaksi);
    
    return ContentService.createTextOutput(JSON.stringify({
      success: true,
      products: dataProduk,
      transactions: dataTransaksi
    })).setMimeType(ContentService.MimeType.JSON);
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    success: false,
    message: "Action tidak valid"
  })).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheetProduk = getOrCreateSheet(ss, "Produk");
  var sheetTransaksi = getOrCreateSheet(ss, "Transaksi");
  
  var postData = JSON.parse(e.postData.contents);
  var action = postData.action;
  
  if (action === "saveTransaction") {
    var trx = postData.transaction;
    
    // Simpan Transaksi baru
    sheetTransaksi.appendRow([
      trx.id,
      trx.date,
      JSON.stringify(trx.items),
      trx.subtotal,
      trx.discount,
      trx.tax,
      trx.total,
      trx.paymentMethod,
      trx.paymentAmount,
      trx.change
    ]);
    
    // Kurangi Stok Produk secara otomatis
    var prodData = sheetProduk.getDataRange().getValues();
    trx.items.forEach(function(item) {
      for (var i = 1; i < prodData.length; i++) {
        if (prodData[i][1] === item.product.code) { // Kolom Kode Produk
          var currentStock = prodData[i][5]; // Kolom Stok (Index ke 5 karena disisipi CostPrice)
          var newStock = Math.max(0, currentStock - item.quantity);
          sheetProduk.getRange(i + 1, 6).setValue(newStock); // Update stok di Sheet
          break;
        }
      }
    });
    
    return ContentService.createTextOutput(JSON.stringify({
      success: true,
      message: "Transaksi berhasil disimpan dan stok telah diperbarui."
    })).setMimeType(ContentService.MimeType.JSON);
  }
  
  if (action === "updateProducts") {
    var products = postData.products;
    
    // Reset data produk dan buat header baru
    sheetProduk.clear();
    sheetProduk.appendRow(["ID", "Code", "Name", "CostPrice", "Price", "Stock", "Category"]);
    
    products.forEach(function(prod) {
      sheetProduk.appendRow([
        prod.id,
        prod.code,
        prod.name,
        prod.costPrice || 0,
        prod.price,
        prod.stock,
        prod.category
      ]);
    });
    
    return ContentService.createTextOutput(JSON.stringify({
      success: true,
      message: "Data produk berhasil diperbarui."
    })).setMimeType(ContentService.MimeType.JSON);
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    success: false,
    message: "Aksi post tidak dikenal."
  })).setMimeType(ContentService.MimeType.JSON);
}

function getOrCreateSheet(ss, name) {
  var sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    if (name === "Produk") {
      sheet.appendRow(["ID", "Code", "Name", "CostPrice", "Price", "Stock", "Category"]);
    } else if (name === "Transaksi") {
      sheet.appendRow(["ID", "Date", "Items", "Subtotal", "Discount", "Tax", "Total", "PaymentMethod", "PaymentAmount", "Change"]);
    }
  }
  return sheet;
}

function readSheetData(sheet) {
  var range = sheet.getDataRange();
  var values = range.getValues();
  var headers = values[0];
  var data = [];
  
  for (var i = 1; i < values.length; i++) {
    var row = {};
    for (var j = 0; j < headers.length; j++) {
      row[headers[j].toLowerCase()] = values[i][j];
    }
    data.push(row);
  }
  return data;
}
